\section{Coordinate Geometry}

\textbf{General Equation:}

\begin{center}
\begin{multicols}{2}
\setlength{\columnseprule}{0pt}
\textbf{Distance Formula}

\bigskip
$D=\sqrt{(x_2-x_1)^2+(y_2-y_1)^2}$

\textbf{Midpoint Formula}

\bigskip
$P=\left(\frac{x_1+x_2}{2},\frac{y_1+y_2}{2}\right)$
\end{multicols}
\end{center}
